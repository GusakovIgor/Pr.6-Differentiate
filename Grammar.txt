G - GENERAL
E - EXPRESSION
T - TERM
D - DEGREE
P - PRIMAL
VAR - VARIABLE
N - NUMBER


+ - >= 1 repeats
* - >= 0 repeats
# - 0 or 1 repeat

G::= E'\0'

E::= T {['+', '-'] T}*
T::= D {['*', '/'] D}*
D::= P {'^' P}*

P::= '(' E ')' | VAR_OR_FUNC | N

VAR_OR_FUNC::= VAR {'(' E ')'}#

VAR::= ['a'-'Z', '_']['a'-'Z', '_', '0'-'9']*

N::= ['0'-'9']+ VAR#
